3.9. Full Script

Script này thực hiện các bước sau:

  1. Tạo ECR repository với tính năng scan-on-push bật sẵn.
  2. Tạo IAM users gồm dev (push image) và ops (pull image).
  3. Gán policy cho từng user để hạn chế quyền theo chức năng.
  4. Build & Push Docker image lên ECR.
  5. Kiểm tra manifest image để đảm bảo hỗ trợ scan.
  6. Scan image và chờ kết quả.
  7. Xuất kết quả scan theo mức độ severity (CRITICAL, HIGH, MEDIUM, LOW, INFORMATIONAL).

Script

#!/usr/bin/env bash
set -euo pipefail

# === Thông số cấu hình ===
REGION="your-region"        
ACCOUNT_ID="your-ID"       
REPO_NAME="your-repo-name"
DEV_USER="dev"
OPS_USER="ops"
DEV_POLICY_NAME="DevPushECRPolicy"
OPS_POLICY_NAME="OpsPullECRPolicy"
IMAGE_TAG="latest"
IMAGE_NAME="${REPO_NAME}:${IMAGE_TAG}"
ECR_REPO_URI="${ACCOUNT_ID}.dkr.ecr.${REGION}.amazonaws.com/${REPO_NAME}"

echo "=== 1. Tạo ECR repository (bật scan-on-push) ==="
aws ecr create-repository   --repository-name "$REPO_NAME"   --image-scanning-configuration scanOnPush=true   --region "$REGION" || echo "Repo đã tồn tại"

echo "=== 2. Tạo IAM users ==="
aws iam create-user --user-name "$DEV_USER" || echo "User dev đã tồn tại"
aws iam create-user --user-name "$OPS_USER" || echo "User ops đã tồn tại"

echo "=== 3. Tạo policy cho dev (push image) ==="
cat > dev-policy.json <<EOF
{
  "Version": "2012-10-17",
  "Statement": [
    {
      "Effect": "Allow",
      "Action": [
        "ecr:GetAuthorizationToken",
        "ecr:BatchCheckLayerAvailability",
        "ecr:PutImage",
        "ecr:InitiateLayerUpload",
        "ecr:UploadLayerPart",
        "ecr:CompleteLayerUpload"
      ],
      "Resource": "arn:aws:ecr:${REGION}:${ACCOUNT_ID}:repository/${REPO_NAME}"
    }
  ]
}
EOF
aws iam create-policy --policy-name "$DEV_POLICY_NAME" --policy-document file://dev-policy.json || echo "Policy dev đã tồn tại"
aws iam attach-user-policy --user-name "$DEV_USER" --policy-arn "arn:aws:iam::${ACCOUNT_ID}:policy/${DEV_POLICY_NAME}"

echo "=== 4. Tạo policy cho ops (pull image) ==="
cat > ops-policy.json <<EOF
{
  "Version": "2012-10-17",
  "Statement": [
    {
      "Effect": "Allow",
      "Action": [
        "ecr:GetAuthorizationToken",
        "ecr:BatchCheckLayerAvailability",
        "ecr:GetDownloadUrlForLayer",
        "ecr:BatchGetImage"
      ],
      "Resource": "arn:aws:ecr:${REGION}:${ACCOUNT_ID}:repository/${REPO_NAME}"
    }
  ]
}
EOF
aws iam create-policy --policy-name "$OPS_POLICY_NAME" --policy-document file://ops-policy.json || echo "Policy ops đã tồn tại"
aws iam attach-user-policy --user-name "$OPS_USER" --policy-arn "arn:aws:iam::${ACCOUNT_ID}:policy/${OPS_POLICY_NAME}"

echo "=== 5. Build & Push Docker image single-platform ==="
aws ecr get-login-password --region "$REGION" | docker login --username AWS --password-stdin "$ECR_REPO_URI"

cat > Dockerfile <<'EOF'
FROM alpine:3.18
CMD ["echo", "Hello from AWS ECR"]
EOF

DOCKER_BUILDKIT=0 docker build --platform linux/amd64 -t "${ECR_REPO_URI}:${IMAGE_TAG}" .
docker push "${ECR_REPO_URI}:${IMAGE_TAG}"

echo "=== 6. Kiểm tra manifest image và scan ==="
MEDIA_TYPE=$(aws ecr batch-get-image   --repository-name "$REPO_NAME"   --image-ids imageTag="$IMAGE_TAG"   --region "$REGION"   --query 'images[0].imageManifestMediaType'   --output text)

echo "Media type: $MEDIA_TYPE"

if [[ "$MEDIA_TYPE" != "application/vnd.docker.distribution.manifest.v2+json" &&       "$MEDIA_TYPE" != "application/vnd.oci.image.manifest.v1+json" ]]; then
  echo "Image media type không hợp lệ để scan. Giá trị nhận được: $MEDIA_TYPE"
  exit 1
fi

DIGEST=$(aws ecr list-images   --repository-name "$REPO_NAME"   --query 'imageIds[?imageTag==`latest`].imageDigest | [0]'   --output text   --region "$REGION")

echo "=== 7. Bắt đầu scan image và chờ kết quả ==="
aws ecr start-image-scan   --repository-name "$REPO_NAME"   --image-id imageDigest="$DIGEST"   --region "$REGION"

while true; do
  STATUS=$(aws ecr describe-image-scan-findings     --repository-name "$REPO_NAME"     --image-id imageDigest="$DIGEST"     --query 'imageScanStatus.status'     --output text     --region "$REGION" 2>/dev/null || echo "IN_PROGRESS")
  echo "Scan status: $STATUS"
  [[ "$STATUS" == "COMPLETE" ]] && break
  sleep 5
done

echo "=== 8. Kết quả scan (tóm tắt) ==="
TOTAL_FINDINGS=$(aws ecr describe-image-scan-findings   --repository-name "$REPO_NAME"   --image-id imageDigest="$DIGEST"   --region "$REGION"   --query 'length(imageScanFindings.findings)' --output text)

CRITICAL_COUNT=$(aws ecr describe-image-scan-findings   --repository-name "$REPO_NAME"   --image-id imageDigest="$DIGEST"   --region "$REGION"   --query "length(imageScanFindings.findings[?severity=='CRITICAL'])" --output text)

HIGH_COUNT=$(aws ecr describe-image-scan-findings   --repository-name "$REPO_NAME"   --image-id imageDigest="$DIGEST"   --region "$REGION"   --query "length(imageScanFindings.findings[?severity=='HIGH'])" --output text)

MEDIUM_COUNT=$(aws ecr describe-image-scan-findings   --repository-name "$REPO_NAME"   --image-id imageDigest="$DIGEST"   --region "$REGION"   --query "length(imageScanFindings.findings[?severity=='MEDIUM'])" --output text)

LOW_COUNT=$(aws ecr describe-image-scan-findings   --repository-name "$REPO_NAME"   --image-id imageDigest="$DIGEST"   --region "$REGION"   --query "length(imageScanFindings.findings[?severity=='LOW'])" --output text)

INFO_COUNT=$(aws ecr describe-image-scan-findings   --repository-name "$REPO_NAME"   --image-id imageDigest="$DIGEST"   --region "$REGION"   --query "length(imageScanFindings.findings[?severity=='INFORMATIONAL'])" --output text || echo 0)

echo "Total findings: ${TOTAL_FINDINGS}"
echo "  CRITICAL: ${CRITICAL_COUNT}"
echo "  HIGH:     ${HIGH_COUNT}"
echo "  MEDIUM:   ${MEDIUM_COUNT}"
echo "  LOW:      ${LOW_COUNT}"
echo "  INFO:     ${INFO_COUNT}"
echo ""

if [[ "${CRITICAL_COUNT}" != "0" || "${HIGH_COUNT}" != "0" ]]; then
  echo "=== Danh sách CRITICAL/HIGH findings ==="
  aws ecr describe-image-scan-findings     --repository-name "$REPO_NAME"     --image-id imageDigest="$DIGEST"     --region "$REGION"     --query "imageScanFindings.findings[?severity=='CRITICAL' || severity=='HIGH'][].{name:name,severity:severity,uri:uri}"     --output table
  echo ""
  echo "Có lỗ hổng mức CRITICAL/HIGH. Kết thúc với exit code 1."
  exit 1
fi

if [[ "${TOTAL_FINDINGS}" != "0" ]]; then
  echo "=== Danh sách các finding (tóm tắt) ==="
  aws ecr describe-image-scan-findings     --repository-name "$REPO_NAME"     --image-id imageDigest="$DIGEST"     --region "$REGION"     --query "imageScanFindings.findings[].{name:name,severity:severity,uri:uri}"     --output table
  echo ""
  echo "Có ${TOTAL_FINDINGS} finding(s). Vui lòng kiểm tra và xử lý."
  exit 0
fi

echo "Không tìm thấy lỗ hổng (0 findings)."
echo "=== Demo hoàn tất ==="
